UP MIGRATION (create users):
  CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
  CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    email VARCHAR NOT NULL UNIQUE,
    name VARCHAR NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
  );

DOWN MIGRATION (create users):
  DROP TABLE IF EXISTS users;

BATCH DATA MIGRATION (PL/pgSQL):
  -- Add column
  ALTER TABLE users ADD COLUMN IF NOT EXISTS display_name VARCHAR;

  SET lock_timeout = '5s';

  -- Backfill in batches
  DO $$
  DECLARE
    batch_size CONSTANT INT := 1000;
    migrated INT := 0;
    row_count INT;
  BEGIN
    LOOP
      WITH batch AS (
        SELECT id, first_name, last_name
        FROM users
        WHERE display_name IS NULL
        ORDER BY id
        LIMIT batch_size
        FOR UPDATE SKIP LOCKED
      )
      UPDATE users u
      SET display_name = CONCAT_WS(' ', b.first_name, b.last_name)
      FROM batch b
      WHERE u.id = b.id;

      GET DIAGNOSTICS row_count = ROW_COUNT;
      EXIT WHEN row_count = 0;
      migrated := migrated + row_count;
      RAISE NOTICE 'Migrated % rows', migrated;
    END LOOP;
  END $$;

COMMON COMMANDS:
  migrate -path migrations -database $DATABASE_URL up      -- apply all
  migrate -path migrations -database $DATABASE_URL down 1  -- revert last
  migrate -path migrations -database $DATABASE_URL version -- show current
  migrate -path migrations -database $DATABASE_URL goto N   -- migrate to version N
  migrate create -ext sql -dir migrations add_column       -- create new
